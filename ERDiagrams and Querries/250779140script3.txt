-- Query 4
SELECT hospitalName, hospitalCode FROM hospital WHERE

-- Query 5
SELECT lName, fName FROM doctor, hospital WHERE
doctor.worksAt = hospital.hospitalCode AND
hospital.hospitalName = "St. Joseph";

-- Query 6
SELECT fName, lName FROM patient WHERE
lName LIKE 'G%';

-- Query 7
SELECT patient.fName, patient.lName FROM patient, seeing, doctor WHERE
patient.OHIPnumber = seeing.OHIPnumber AND
doctor.licenseNumber = seeing.licenseNumber AND
doctor.lName = "Clooney";

-- Query 8
SELECT hospitalName, city, lName FROM hospital, doctor WHERE
hospital.headIDnumber = doctor.licenseNumber;

-- Query 9
SELECT SUM(numBeds) AS "TotalBeds" FROM hospital;

-- Query 10
SELECT patient.fName, doctor.fName FROM patient, doctor, hospital, seeing WHERE
patient.OHIPnumber = seeing.OHIPnumber AND
doctor.licenseNumber = seeing.licenseNumber AND
hospital.headIDnumber = doctor.licenseNumber;

-- Query 11
SELECT fName, lName FROM doctor, hospital WHERE
doctor.specialty = "Surgeon" AND
doctor.worksAt = hospital.hospitalCode AND
hospital.hospitalName = "Victoria";

-- Query 12
SELECT DISTINCT fName FROM doctor, seeing WHERE
doctor.licenseNumber NOT IN (SELECT licenseNumber FROM seeing);

-- Query 13
SELECT fName, lName FROM doctor, seeing WHERE
doctor.licenseNumber = seeing.licenseNumber GROUP BY doctor.licenseNumber HAVING COUNT(doctor.licenseNumber) >1;
